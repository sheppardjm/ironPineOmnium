**Client-side script (`<script>` tag, NOT `is:inline`):**

The script runs on DOMContentLoaded and does three things: decode payload, render preview, handle form.

**Part A — Payload decode:**

```javascript
function fromBase64url(encoded) {
  let b = encoded.replace(/-/g, "+").replace(/_/g, "/");
  while (b.length % 4) b += "=";
  return JSON.parse(atob(b));
}

document.addEventListener("DOMContentLoaded", () => {
  const params = new URLSearchParams(window.location.search);
  const encoded = params.get("payload");

  if (!encoded) {
    window.location.href = "/submit";
    return;
  }

  let payload;
  try {
    payload = fromBase64url(encoded);
  } catch (_) {
    window.location.href = "/submit";
    return;
  }

  renderPreview(payload);
  populateHiddenFields(payload);
  setupForm();
});
```

**Part B — Score preview rendering (`renderPreview` function):**

Format time helper (same logic as scoring.ts `formatDuration`):
```javascript
function formatTime(totalSeconds) {
  const h = Math.floor(totalSeconds / 3600);
  const m = Math.floor((totalSeconds % 3600) / 60);
  const s = totalSeconds % 60;
  return [h, m, s].map(v => String(v).padStart(2, "0")).join(":");
}
```

**Part D — Form handling (`setupForm`):**

```javascript
function setupForm() {
  const form = document.getElementById("confirm-form");
  form.addEventListener("submit", (e) => {
    e.preventDefault();

    const nameInput = document.getElementById("rider-name");
    const categorySelect = document.getElementById("rider-category");
    const name = nameInput.value.trim();
    const category = categorySelect.value;

    // Reset custom validity
    nameInput.setCustomValidity("");
    categorySelect.setCustomValidity("");

    if (!name) {
      nameInput.setCustomValidity("Please enter your display name.");
      nameInput.reportValidity();
      return;
    }

    if (!category) {
      categorySelect.setCustomValidity("Please select your category.");
      categorySelect.reportValidity();
      return;
    }

    // Phase 7 will add the actual submit-result POST here.
    // For now, show a placeholder confirmation.
    const btn = document.getElementById("confirm-btn");
    btn.textContent = "Submission ready";
    btn.disabled = true;
  });

  // Clear custom validity on input to allow re-submission
  document.getElementById("rider-name").addEventListener("input", function() {
    this.setCustomValidity("");
  });
  document.getElementById("rider-category").addEventListener("change", function() {
    this.setCustomValidity("");
  });
}
```
