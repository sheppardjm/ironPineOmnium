// Strava identity display
const identityEl = document.getElementById("athlete-identity");
if (identityEl) {
  const first = (payload.athleteFirstname ?? "").trim();
  const last = (payload.athleteLastname ?? "").trim();
  const fullName = [first, last].filter(Boolean).join(" ");
  if (fullName) {

}
